    def __init__(self, name, speed):
        self.name = name
        self.speed = speed
        self.sleep = True
        self.place = False
    def __str__(self):
        return "Barber "+str(self.name)+" speed of work is "+str(self.speed)+"%"
    def work(self, client, n):
        client.hair=n
        
        return (client.hair-n)/self.speed
